\documentclass{standalone}
\usepackage{pgfplots}
\pgfplotsset{compat=1.18}

\begin{document}

\begin{tikzpicture}
    \begin{axis}
    [
        axis lines = middle,
        xlabel = $x$,
        ylabel = {$f(x)$},
        samples=100,
        xmin=-0.5, xmax=4.5,
        ymin=-0.5, ymax=1
    ]
                
    \addplot[color=blue, thick, domain=0:4]{1/4};

    % Segunda parte: f(x) = 0 en otros casos
    \addplot[color=red, thick, domain=-0.5:0]{0};
    \addplot[color=red, thick, domain=4:4.5]{0};

    \addplot[only marks, mark=o, mark size=2pt, color=blue] coordinates {(0,1/4)};

    \addplot[only marks, mark=o, mark size=2pt, color=blue] coordinates {(4,1/4)};

    \addplot[only marks, mark=*, mark size=2pt, color=blue] coordinates {(0,0)};

    \addplot[only marks, mark=*, mark size=2pt, color=blue] coordinates {(4,0)};
    \end{axis}
\end{tikzpicture}

\end{document}
